#!/bin/bash

source /home/xx/miniconda3/etc/profile.d/conda.sh
conda deactivate
conda activate base

export HF_ENDPOINT=https://hf-mirror.com

if [ "$#" -ne 5 ]; then
    echo "Usage: $0 <type> <name> <local_dir> <cache_dir>  <if_hftrans>"
    exit 1
fi

type=$1
name=$2
local_dir=$3
cache_dir=$4
if_hftrans=$5

if [ "$if_hftrans" -eq 1 ]; then
    export HF_HUB_ENABLE_HF_TRANSFER=1
    export
    huggingface-cli download \
        --repo-type "${type}" \
        "${name}" \
        --local-dir "${local_dir}" \
        --cache-dir "${cache_dir}" \
        --local-dir-use-symlinks False
else
    export
    huggingface-cli download \
        --repo-type "${type}" \
        "${name}" \
        --local-dir "${local_dir}" \
        --cache-dir "${cache_dir}" \
        --local-dir-use-symlinks False \
        --resume-download
fi
